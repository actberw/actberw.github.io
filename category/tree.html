二叉排序树(binary search tree)
Tue 18 November 2014

typedef struct bst_node {    
    char data;    
    struct bst_node *lchild,*rchild;

} bst_node, *bst_tree;

int search(bst_tree root, char data, bst_node ** parent) {
    bst_node p = root, *parent = NULL; 
    while (p != NULL) {
        if (p->data == data) return 1;
        *parent = p;
        if (data < p->data)  p = p->lchild;
        else  p = p->rchild;
    }
    return 0;
}
