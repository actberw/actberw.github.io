二进制表示1的个数

# 解法1
int count(int v) {
    int num = 0;
    while (v) {
        if (v % 2 == 1) {
            num++;
        }
        v /=2;
    }
    return num;
}

# 解法2
int count(int v) {
    int num = 0;
    while (v) {
        num += v & 0x1;
        v >>= 1;
    }
    return num;
}

# 解法 3
int count(int v) {
    int num = 0;
    while (v) {
        v &= (v-1);
        num++;
    }
    return num;
}

反转整数二进制表示

# 解法1
int bit_reverse(int v) {
    int size = sizeof(v) * 8, result=0, i;
    for (i = 0; i < size; v >>= 1) {
        result = (result << 1) | (v & 0x01);
    }
    return result;
}

二分查找32位整数的前导0个数

Hacker's Delight上的代码, 程序思想是二分查找.

int nlz(unsigned x) {
   int n;
   if (x == 0) return(32);
   n = 1;
   if ((x >> 16) == 0) {n = n +16; x = x <<16;}
   if ((x >> 24) == 0) {n = n + 8; x = x << 8;}
   if ((x >> 28) == 0) {n = n + 4; x = x << 4;}
   if ((x >> 30) == 0) {n = n + 2; x = x << 2;}
   n = n - (x >> 31);
   return n;
}

判断二进制是否为回文数

参见十进制数是否为回文数, 二进制比较不能直接反转比较，因为反转后可能是个负数.

is_palindrome(int x) {
    int a = x>>1, mask = 1;
    while(a) {
        mask <<= 1;
        a >>= 1;
    }
    while (a) {     //能否用a判断?
        if ((x & mask) ^ (x & 0x1) != 0x0) return 0;
        x &=(~mask); //去高位
        x >>= 1;     //去低位
        mask >>= 2; 
    }
    return 1;
}
